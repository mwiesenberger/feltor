348inline JsonType file2Json(std::string filename, enum comments comm =
349 file::comments::are_discarded, enum error err = file::error::is_throw)
350{
351 std::ifstream isI( filename);
352 if( !isI.good())
353 {
354 std::string message = "\nAn error occured while parsing "+filename+"\n";
355 message += "*** File does not exist! *** \n\n";
356 if( err == error::is_throw)
357 throw std::runtime_error( message);
358 else if (err == error::is_warning)
359 std::cerr << "WARNING: "<<message<<std::endl;
360 else
361 {
362 }
363 return JsonType();
364 }
365 JsonType js;
366#ifdef DG_USE_JSONHPP
367 bool ignore_comments = false, allow_exceptions = false;
368 if ( comm == file::comments::are_discarded)
369 ignore_comments = true;
370 if ( err == error::is_throw)
371 allow_exceptions = true; //throws nlohmann::json::parse_error
372
373 js = nlohmann::json::parse( isI, nullptr, allow_exceptions, ignore_comments);
374 if( !allow_exceptions && err == error::is_warning && js.is_discarded())
375 {
376 std::string message = "An error occured while parsing "+filename+"\n";
377 std::cerr << "WARNING: "<<message<<std::endl;
378 }
379#else
380 Json::CharReaderBuilder parser;
381 if( comments::are_forbidden == comm )
382 Json::CharReaderBuilder::strictMode( &parser.settings_);
383 else if( comments::are_discarded == comm )
384 {
385 Json::CharReaderBuilder::strictMode( &parser.settings_);
386 // workaround for a linker bug in jsoncpp from package manager
387 JsonType js_true (true);
388 JsonType js_false (false);
389 parser.settings_["allowComments"].swap( js_true);
390 parser.settings_["collectComments"].swap(js_false);
391 }
392 else
393 Json::CharReaderBuilder::setDefaults( &parser.settings_);
394
395 std::string errs;
396 if( !parseFromStream( parser, isI, &js, &errs) )
397 {
398 std::string message = "An error occured while parsing "+filename+"\n"+errs;
399 if( err == error::is_throw)
400 throw std::runtime_error( message);
401 else if (err == error::is_warning)
402 std::cerr << "WARNING: "<<message<<std::endl;
403 else
404 {
405 }
406 }
407#endif
408 return js;
409}

432inline JsonType string2Json(std::string input, enum comments comm = file::comments::are_discarded, enum error err = file::error::is_throw)
433{
434 JsonType js;

438 ignore_comments = true;

440 allow_exceptions = true; //throws nlohmann::json::parse_error
441
442
443 js = nlohmann::json::parse( input, nullptr, allow_exceptions, ignore_comments);
444 if( !allow_exceptions && err == error::is_warning && js.is_discarded())
445 {
446 std::string message = "An error occured while parsing \n";
447 std::cerr << "WARNING: "<<message<<std::endl;
448 }
449 return js;
450
451#else
452
453 Json::CharReaderBuilder parser;
454 if( comments::are_forbidden == comm )
455 Json::CharReaderBuilder::strictMode( &parser.settings_);
456 else if( comments::are_discarded == comm )
457 {
458 Json::CharReaderBuilder::strictMode( &parser.settings_);
459 parser.settings_["allowComments"] = true;
460 parser.settings_["collectComments"] = false;
461 }

463 Json::CharReaderBuilder::setDefaults( &parser.settings_);

469 if( err == error::is_throw)

471 else if (err == error::is_warning)
472 std::cerr << "WARNING: "<<errs<<std::endl;
473 else
