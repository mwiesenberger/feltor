23template<class T>
24T gcd( T a, T b)
25{
26 T r2 = std::max(a,b);
27 T r1 = std::min(a,b);
28 while( r1!=0)
29 {
30 r2 = r2%r1;
31 std::swap( r1, r2);
32 }
33 return r2;
34}
