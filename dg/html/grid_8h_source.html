128 host_vector abscissas(unsigned u=0) const
129 {
130 // On getting n,N,x0,x1 given the abscissas:
131 // Unfortunately, we cannot invert this to binary precision
132 // at least not with our pyfeltor implementation
133 if( u >= Nd)
134 throw Error( Message(_ping_)<<"u>=Nd not allowed! You typed: "
135 <<u<<" while Nd is "<<Nd);
136 host_vector abs(m_n[u]*m_N[u]);
137 real_type hu = h(u);
138 auto aa = dg::DLT<real_type>::abscissas(m_n[u]);
139 for( unsigned i=0; i<m_N[u]; i++)
140 for( unsigned j=0; j<m_n[u]; j++)
141 {
142 real_type xmiddle = DG_FMA( hu, (real_type)(i), m_x0[u]);
143 real_type h2 = hu/2.;
144 real_type absj = 1.+aa[j];
145 abs[i*m_n[u]+j] = DG_FMA( h2, absj, xmiddle);
146 }
147 return abs;
148 }
155 host_vector weights(unsigned u=0) const
156 {
157 if( u >= Nd)
158 throw Error( Message(_ping_)<<"u>Nd not allowed! You typed: "<<u<<" while Nd is "<<Nd);
159 host_vector v( m_n[u]*m_N[u]);
160 auto ww = dg::DLT<real_type>::weights(m_n[u]);
161 real_type hu = h(u);
162 for( unsigned i=0; i<m_N[u]; i++)
163 for( unsigned j=0; j<m_n[u]; j++)
164 v[i*m_n[u] + j] = hu/2.*ww[j];
165 return v;
166 }
