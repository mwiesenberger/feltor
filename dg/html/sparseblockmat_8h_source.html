340cusp::coo_matrix<int, real_type, cusp::host_memory> EllSparseBlockMat<real_type, Vector>::asCuspMatrix() const
341{
342 cusp::array1d<real_type, cusp::host_memory> values;
343 cusp::array1d<int, cusp::host_memory> row_indices;
344 cusp::array1d<int, cusp::host_memory> column_indices;
345 for( int s=0; s<left_size; s++)
346 for( int i=0; i<num_rows; i++)
347 for( int k=0; k<n; k++)
348 for( int j=right_range[0]; j<right_range[1]; j++)
349 {
350 int I = ((s*num_rows + i)*n+k)*right_size+j;
351 for( int d=0; d<blocks_per_line; d++)
352 for( int q=0; q<n; q++) //multiplication-loop
353 {
354 row_indices.push_back(I);
355 int J = cols_idx[i*blocks_per_line+d];
356 if ( J == invalid_index)
357 continue;
358 column_indices.push_back(
359 ((s*num_cols + J)*n+q)*right_size+j);
360 values.push_back(data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q]);
361 }
362 }
363 cusp::coo_matrix<int, real_type, cusp::host_memory> A(
364 total_num_rows(), total_num_cols(), values.size());
365 A.row_indices = row_indices;
366 A.column_indices = column_indices;
367 A.values = values;
368 return A;
369}
