702 }

716 dg::blas2::symv( m_jfactor, m_jumpY, x, 0., m_tempy);
717 if( m_addJumpZ)
718 {
719 dg::blas2::symv( m_jfactor, m_jumpZ, x, 0., m_tempz);
720 dg::tensor::multiply3d(m_sigma, m_chi, m_tempx, m_tempy,
721 m_tempz, 0., m_tempx, m_tempy, m_tempz);
722 }

730 }
