433template< class T>
434T lu_pivot( dg::SquareMatrix<T>& m, std::vector<unsigned>& p)
435{
436 //from numerical recipes
437 T pivot, determinant=(T)1;
438 unsigned pivotzeile, numberOfSwaps=0;
439 const size_t n = m.size();
440 p.resize( n);
441 for( size_t j = 0; j < n; j++) //gehe Spalten /Diagonale durch
442 {
443 //compute upper matrix except for the diagonal element (the pivot)
444 for( size_t i = 0; i< j; i++)
445 {
446 thrust::host_vector<T> mik(i), mkj(i);
447 for( size_t k=0; k<i; k++)
448 mik[k] = m(i,k), mkj[k] = m(k,j);
449 m(i,j) -= dg::blas1::dot( mik, mkj);
450 }
451 //compute canditates for pivot elements
452 for( size_t i = j; i< n; i++)
453 {
454 thrust::host_vector<T> mik(j), mkj(j);
455 for( size_t k=0; k<j; k++)
456 mik[k] = m(i,k), mkj[k] = m(k,j);
457 m(i,j) -= dg::blas1::dot( mik, mkj);
458 }
459 //search for absolute maximum of pivot candidates
460 pivot = m(j,j);
461 pivotzeile = j;
462 for( size_t i = j+1; i < n; i++)
463 if( fabs( m(i,j)) > fabs(pivot))
464 {
465 pivot = m(i,j), pivotzeile = i;
466 }
467
468 if( fabs(pivot) > 1e-15 )
469 {
470 if( pivotzeile != j)
471 {
472 m.swap_lines( pivotzeile, j);
473 numberOfSwaps++;
474 }
475 p[j] = pivotzeile;
476 //divide all elements below the diagonal by the pivot to get the lower matrix
477 for( size_t i=j+1; i<n; i++)
478 m(i,j) /= pivot;
479 determinant*=m(j,j);
480
481 }
482 else
483 throw std::runtime_error( "Matrix is singular!!");
484 }
485 if( numberOfSwaps % 2 != 0)
486 determinant*=-1.;
487 return determinant;
488
489}

506template<class T>
507dg::SquareMatrix<T> inverse( const dg::SquareMatrix<T>& in)
508{
509 dg::SquareMatrix<T> out(in);
510 const unsigned n = in.size();
511 std::vector<unsigned> pivot( n);
512 dg::SquareMatrix<T> lu(in);
513 T determinant = lu_pivot( lu, pivot);
514 if( fabs(determinant ) == 0)
515 throw std::runtime_error( "Determinant zero!");
516 for( unsigned i=0; i<n; i++)
517 {
518 std::vector<T> unit(n, 0);
519 unit[i] = 1;
520 lu_solve( lu, pivot, unit);
521 for( unsigned j=0; j<n; j++)
522 out(j,i) = unit[j];
523 }
524 return out;
525}

547template<class T>
548void lu_solve( const dg::SquareMatrix<T>& lu, const std::vector<unsigned>& p, std::vector<T>& b)
549{
550 assert(p.size() == lu.size() && p.size() == b.size());
551 const size_t n = p.size();
552 // Vorwärtseinsetzen
553 for( size_t i = 0; i<n; i++)
554 {
555 //mache Zeilentausch
556 std::swap( b[ p[i] ], b[i]);
557 thrust::host_vector<T> lui(i), bi(i);
558 for( size_t j = 0; j < i; j++)
559 lui[j] = lu(i,j), bi[j] = b[j];
560 b[i] -= dg::blas1::dot( lui, bi);
561 }
562 // Rückwärtseinsetzen
563 for( int i = n-1; i>=0; i--)
564 {
565 thrust::host_vector<T> lui(n-(i+1)), bi(n-(i+1));
566 for( size_t j = i+1; j < n; j++)
567 lui[j-(i+1)] = lu(i,j), bi[j-(i+1)] = b[j];
568 b[i] -= dg::blas1::dot( lui, bi);
569 b[i] /= lu(i,i);
570 }
571}
