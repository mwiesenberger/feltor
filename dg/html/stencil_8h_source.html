19template<class real_type>
20 void set_boundary(
21 cusp::array1d<real_type, cusp::host_memory>& values,
22 cusp::array1d<int, cusp::host_memory>& column_indices,
23 dg::bc bcx,
24 int num_cols)
25{
26 for( unsigned k=0; k<values.size(); k++)
27 {
28 if( column_indices[k] < 0 )
29 {
30 if( bcx == dg::NEU || bcx == dg::NEU_DIR)
31 column_indices[k] = -(column_indices[k]+1);
32 else if( bcx == dg::DIR || bcx == dg::DIR_NEU)
33 {
34 column_indices[k] = -(column_indices[k]+1);
35 values[k] *= -1;
36 }
37 else if( bcx == dg::PER)
38 column_indices[k] += num_cols;
39 }
40 else if( column_indices[k] >= num_cols)
41 {
42 if( bcx == dg::NEU || bcx == dg::DIR_NEU)
43 column_indices[k] = 2*num_cols-1-column_indices[k];
44 else if( bcx == dg::DIR || bcx == dg::NEU_DIR)
45 {
46 column_indices[k] = 2*num_cols-1-column_indices[k];
47 values[k] *= -1;
48 }
49 else if( bcx == dg::PER)
50 column_indices[k] -= num_cols;
51 }
52 }
53}

55template<class real_type>
56cusp::csr_matrix<int, real_type, cusp::host_memory> window_stencil(
57 unsigned stencil_size,
58 const RealGrid1d<real_type>& local,
59 const RealGrid1d<real_type>& global,
60 dg::bc bcx)
61{
62 cusp::array1d<real_type, cusp::host_memory> values;
63 cusp::array1d<int, cusp::host_memory> row_offsets;
64 cusp::array1d<int, cusp::host_memory> column_indices;
65
66 unsigned num_rows = local.size();
67 unsigned num_cols = global.size();
68 unsigned radius = stencil_size/2;
69 int L0 = round((local.x0() - global.x0())/global.h())*global.n();
70
71 row_offsets.push_back(0);
72 for( unsigned k=0; k<num_rows; k++)
73 {
74 row_offsets.push_back(stencil_size + row_offsets[k]);
75 for( unsigned l=0; l<stencil_size; l++)
76 {
77 column_indices.push_back( L0 + (int)(k + l) - (int)radius);
78 values.push_back( 1.0);
79 }
80 }
81 set_boundary( values, column_indices, bcx, num_cols);
82
83 cusp::csr_matrix<int, real_type, cusp::host_memory> A(
84 num_rows, num_cols, values.size());
85 A.row_offsets = row_offsets;
86 A.column_indices = column_indices;
87 A.values = values;
88 return A;
89}

92template<class real_type>
93cusp::csr_matrix<int, real_type, cusp::host_memory> limiter_stencil(
94 const RealGrid1d<real_type>& local,
95 const RealGrid1d<real_type>& global,
96 dg::bc bcx)
97{
98 cusp::array1d<real_type, cusp::host_memory> values;
99 cusp::array1d<int, cusp::host_memory> row_offsets;
100 cusp::array1d<int, cusp::host_memory> column_indices;
101
102 unsigned num_rows = local.size();
103 unsigned num_cols = global.size();
104 int L0 = round((local.x0() - global.x0())/global.h())*global.n();
105 // We need the first two lines of forward
106 dg::SquareMatrix<real_type> forward = dg::DLT<real_type>::forward(local.n());
107 // and the second column of backward
108 dg::SquareMatrix<real_type> backward = dg::DLT<real_type>::backward(local.n());
109 if( global.n() == 1)
110 throw dg::Error( dg::Message(_ping_) << "Limiter stencil not possible for n==1!");
111
112
113 row_offsets.push_back( 0);
114 for( unsigned k=0; k<local.N(); k++)
115 {
116 row_offsets.push_back(row_offsets[row_offsets.size()-1] + 3*local.n());
117 for( unsigned j=1; j<local.n(); j++)
118 row_offsets.push_back(row_offsets[row_offsets.size()-1]);
119 // Order is important
120 for( unsigned j=0; j<local.n(); j++)
121 {
122 column_indices.push_back( L0 + (int)((k-1)*global.n() + j) );
123 values.push_back( forward(0, j ));
124 }
125 for( unsigned j=0; j<local.n(); j++)
126 {
127 column_indices.push_back( L0 + (int)(k*global.n() + j ));
128 values.push_back( forward(1, j ));
129 }
130 for( unsigned j=0; j<local.n(); j++)
131 {
132 column_indices.push_back( L0 + (int)((k+1)*global.n() + j));
133 values.push_back( backward(j, 1) );
134 }
135 }
136 assert( row_offsets.size() == num_rows+1);
137 set_boundary( values, column_indices, bcx, num_cols);
138
139 cusp::csr_matrix<int, real_type, cusp::host_memory> A(
140 num_rows, num_cols, values.size());
141 A.row_offsets = row_offsets;
142 A.column_indices = column_indices;
143 A.values = values;
144 return A;
145}

147template<class real_type>
148cusp::csr_matrix< int, real_type, cusp::host_memory> identity_matrix( const RealGrid1d<real_type>& local, const RealGrid1d<real_type>& global)
149{
150 cusp::csr_matrix<int,real_type,cusp::host_memory> A( local.size(),
151 global.size(), local.size());
152 int L0 = round((local.x0() - global.x0())/global.h())*global.n();
153 A.row_offsets[0] = 0;
154 for( unsigned i=0; i<local.size(); i++)
155 {
156 A.row_offsets[i+1] = 1 + A.row_offsets[i];
157 A.column_indices[i] = L0 + i;
158 A.values[i] = 1.;
159 }
160 return A;
161}
