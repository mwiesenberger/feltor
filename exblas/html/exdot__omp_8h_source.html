44inline void ReductionStep(int step, int64_t * acc1, int64_t * acc2,
45 int volatile * ready)
46{
47#ifndef _WITHOUT_VCL
48 _mm_prefetch((char const*)ready, _MM_HINT_T0);
49 // Wait for thread 2 to be ready
50 while(*ready < step) {
51 // wait
52 _mm_pause();
53 }
54#endif//_WITHOUT_VCL
55 int imin = IMIN, imax = IMAX;
56 Normalize( acc1, imin, imax);
57 imin = IMIN, imax = IMAX;
58 Normalize( acc2, imin, imax);
59 for(int i = IMIN; i <= IMAX; ++i) {
60 acc1[i] += acc2[i];
61 }
62}

71inline void Reduction(unsigned int tid, unsigned int tnum, std::vector<int32_t>& ready,
72 std::vector<int64_t>& acc, int const linesize)
73{
74 // Custom tree reduction
75 for(unsigned int s = 1; (unsigned)(1 << (s-1)) < tnum; ++s)
76 {
77 // 1<<(s-1) = 0001, 0010, 0100, ... = 1,2,4,8,16,...
78 int32_t volatile * c = &ready[tid * linesize];
79 ++*c; //set: ready for level s
80#ifdef _WITHOUT_VCL
81#pragma omp barrier //all threads are ready for level s
82#endif
83 if(tid % (1 << s) == 0) { //1<<s = 2,4,8,16,32,...
84 //only the tid thread executes this block, tid2 just sets ready
85 unsigned int tid2 = tid | (1 << (s-1)); //effectively adds 1, 2, 4,...
86 if(tid2 < tnum) {
87 ReductionStep(s, &acc[tid*BIN_COUNT], &acc[tid2*BIN_COUNT],
88 &ready[tid2 * linesize]);
89 }
90 }
91 }
92}
