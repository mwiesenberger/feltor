75inline void AccumulateWord( int64_t *accumulator, int i, int64_t x, int stride = 1) {
76 // With atomic superacc updates
77 // accumulation and carry propagation can happen in any order,
78 // as long as addition is atomic
79 // only constraint is: never forget an overflow bit
80 //assert(i >= 0 && i < BIN_COUNT);
81 unsigned char overflow;
82 int64_t carry = x;
83 int64_t carrybit;
84 int64_t oldword = xadd(accumulator[i * stride], x, overflow);
85 // To propagate over- or underflow
86 while (overflow) {
87 // Carry or borrow
88 // oldword has sign S
89 // x has sign S
90 // accumulator[i] has sign !S (just after update)
91 // carry has sign !S
92 // carrybit has sign S
93 carry = (oldword + carry) >> DIGITS; // Arithmetic shift
94 bool s = oldword > 0;
95 carrybit = (s ? 1l << KRX : -1l << KRX);
96
97 // Cancel carry-accumulatorve bits
98 xadd(accumulator[i * stride], (int64_t) -(carry << DIGITS), overflow);
99 if (TSAFE && (s ^ overflow)){ //MW: TSAFE is always 0
100 // (Another) overflow of sign S
101 carrybit *= 2;
102 }
103 carry += carrybit;
104
105 ++i;
106 if (i >= BIN_COUNT){
107 //status = Overflow;
108 return;
109 }
110 oldword = xadd(accumulator[i * stride], carry, overflow);
111 }
112}

123inline void Accumulate( int64_t* accumulator, double x, int stride = 1) { //transposed accumulation
124 if (x == 0)
125 return;
126 //MW: This assert does not help very much in finding out where the nan originates
127 //assert( !std::isnan(x) && "Detected NaN in dot product!!");
128
129 int e;
130 frexp(x, &e); //extract the exponent of x (lies in -1024;1023 ?)
131 int exp_word = e / DIGITS; // Word containing MSbit
132 int iup = exp_word + F_WORDS; //can be at most 18 + 20
133
134 double xscaled = ldexp(x, -DIGITS * exp_word);
135
136 int i;
137 for (i = iup; i>=0 && xscaled != 0; --i) { //MW: i>=0 protects agains NaN
138 double xrounded = rint(xscaled);
139 int64_t xint = (int64_t) xrounded;
140 AccumulateWord(accumulator, i, xint, stride);
141
142 xscaled -= xrounded;
143 xscaled *= DELTASCALE;
144 }
145}

161__device__
162inline int Normalize( int64_t *accumulator, int& imin, int& imax, int stride = 1) {
163 int64_t carry_in = accumulator[(imin)*stride] >> DIGITS;
164 accumulator[(imin)*stride] -= carry_in << DIGITS;
165 int i;
166 // Sign-extend all the way
167 for (i = imin + 1; i < BIN_COUNT; ++i) {
168 accumulator[i*stride] += carry_in;
169 int64_t carry_out = accumulator[i*stride] >> DIGITS; // Arithmetic shift
170 accumulator[i*stride] -= (carry_out << DIGITS);
171 carry_in = carry_out;
172 }
173 imax = i - 1;
174
175 // Do not cancel the last carry to avoid losing information
176 accumulator[imax*stride] += carry_in << DIGITS;
177
178 return carry_in < 0;
179}

192inline double Round( int64_t * accumulator) {
193 int imin = IMIN;
194 int imax = IMAX;
195 int negative = Normalize(accumulator, imin, imax);
196
197 // Find leading word
198 int i;
199 // Skip zeroes
200 for (i = imax; i >= imin && accumulator[i] == 0; --i) {
201 // MW: note that i >= imin has to come *before* accumulator[i]
202 // else it is possible that accumulator[-1] is accessed
203 }
204 if (negative) {
205 // Skip ones
206 for(; i >= imin && (accumulator[i] & ((1ll << DIGITS) - 1)) == ((1ll << DIGITS) - 1);--i) {
207 }
208 }
209 if (i < 0) {
210 return 0.0;
211 }
212
213 int64_t hiword = negative ? ((1ll << DIGITS) - 1) - accumulator[i] : accumulator[i];
214 double rounded = (double)hiword;
215 double hi = ldexp(rounded, (i - F_WORDS) * DIGITS);
216 if (i == 0) {
217 return negative ? -hi : hi; // Correct rounding achieved
218 }
219 hiword -= (int64_t) rint(rounded);
220 double mid = ldexp((double) hiword, (i - F_WORDS) * DIGITS);
221
222 // Compute sticky
223 int64_t sticky = 0;
224 for (int j = imin; j != i - 1; ++j) {
225 sticky |= negative ? ((1ll << DIGITS) - accumulator[j]) : accumulator[j];
226 }
227
228 int64_t loword = negative ? ((1ll << DIGITS) - accumulator[i - 1]) : accumulator[i - 1];
229 loword |= !!sticky;
230 double lo = ldexp((double) loword, (i - 1 - F_WORDS) * DIGITS);
231
232 // Now add3(hi, mid, lo)
233 // No overlap, we have already normalized
234 if (mid != 0) {
235 lo = OddRoundSumNonnegative(mid, lo);
236 }
237 // Final rounding
238 hi = hi + lo;
239 return negative ? -hi : hi;
240}
