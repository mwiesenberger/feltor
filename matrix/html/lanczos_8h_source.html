206 void normMbVy( MatrixType&& A,
207 const DiaMatrixType& T,
208 const ContainerType0& y,
209 ContainerType1& x,
210 const ContainerType2& b, value_type bnorm)
211 {
212 dg::blas1::copy(0., x);
213 if( 0 == bnorm )
214 {
215 return;
216 }
217 dg::blas1::axpby(1./bnorm, b, 0.0, m_v); //m_v[1] = b/||b||
218 dg::blas1::copy(0., m_vm);
219 // check if (potentially) all higher elements in y are zero
220 unsigned less_iter = 0;
221 for( unsigned i=0; i<y.size(); i++)
222 if( y[i] != 0)
223 less_iter = i+1;
224 dg::blas1::axpby( y[0]*bnorm, m_v, 1., x); //Compute b= |b| V y
225
226 for ( unsigned i=0; i<less_iter-1; i++)
227 {
228 dg::blas2::symv( std::forward<MatrixType>(A), m_v, m_vp);
229 dg::blas1::axpbypgz(
230 -T.values(i,0)/T.values(i,2), m_vm,
231 -T.values(i,1)/T.values(i,2), m_v,
232 1.0/T.values(i,2), m_vp);
233 dg::blas1::axpby( y[i+1]*bnorm, m_vp, 1., x); //Compute b= |b| V y
234 m_vm.swap( m_v);
235 m_v.swap( m_vp);
236 }

348 template<class UnaryOp>
349 value_type compute_universal_error( const HDiaMatrix& TH, unsigned iter,
350 unsigned q, UnaryOp f, HVec& yH)
351 {
352 unsigned new_iter = iter + 1 + q;
353 set_iter( iter+1);
354 HDiaMatrix THtilde( new_iter, new_iter, 3*new_iter-2, 3);
355 THtilde.diagonal_offsets[0] = -1;
356 THtilde.diagonal_offsets[1] = 0;
357 THtilde.diagonal_offsets[2] = 1;
358 for( unsigned u=0; u<iter+1; u++)
359 {
360 THtilde.values(u,0) = TH.values(u,0);
361 THtilde.values(u,1) = TH.values(u,1);
362 THtilde.values(u,2) = TH.values(u,2);
363 }
364 for( unsigned u=1; u<=q; u++)
365 {
366 THtilde.values( iter+u, 0) = u==1 ? TH.values(iter,2) :
367 TH.values( iter+1-u, 1);
368 THtilde.values( iter+u, 1) = TH.values( iter-u, 1);
369 THtilde.values( iter+u, 2) = TH.values( iter-u, 0);
370 }
371 yH = f( TH);
372 HVec yHtilde = f( THtilde);
373 for( unsigned u=0; u<yH.size(); u++)
374 yHtilde[u] -= yH[u];
375 value_type norm = dg::fast_l2norm( yHtilde);
376 return norm;
377 }
